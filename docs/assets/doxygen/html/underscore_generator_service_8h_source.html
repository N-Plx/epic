107 void printAndSaveGenerationConfiguration() {
108
109 GenerationInformation generationInformation;
110
111 generationInformation.setGeneratorName("EPIC");
112 generationInformation.setGeneratorVersion(
113 ElemUtils::Formatter() << EPIC_VERSION_MAJOR << "."
114 << EPIC_VERSION_MINOR << "." << EPIC_VERSION_PATCH);
115 generationInformation.setDescription(
116 m_scenarioDate + " | " + m_scenarioDescription);
117
118 generationInformation.setServiceName(getClassName());
119 generationInformation.setNEvents(m_generalConfiguration.getNEvents());
120 generationInformation.setIntegratedCrossSection(
121 m_pEventGeneratorModule->getIntegral());
122 generationInformation.setGenerationDate(getCurrentDate());
123
124 generationInformation.setInitializationTime(1.E-6*std::chrono::duration_cast<std::chrono::microseconds>(m_debugTimeInitialization.second- m_debugTimeInitialization.first).count());
125 generationInformation.setGenerationTime(1.E-6*std::chrono::duration_cast<std::chrono::microseconds>(m_debugTimeGeneration.second- m_debugTimeGeneration.first).count());
126
127 m_pWriterModule->saveGenerationInformation(generationInformation);
128
129 info(__func__,
130 ElemUtils::Formatter() << "Generation information:\n"
131 << generationInformation.toString() << '\n');
132 }

334 void getEventGeneratorModuleFromTask(const MonteCarloTask &task) {
335
336 // check if available
337 if (ElemUtils::StringUtils::equals(
338 task.getGeneratorConfiguration().getModuleType(),
339 EventGeneratorModule::EVENT_GENERATOR_MODULE_CLASS_NAME)) {
340
341 // configure
342 m_pEventGeneratorModule =
343 Epic::getInstance()->getModuleObjectFactory()->newEventGeneratorModule(
344 task.getGeneratorConfiguration().getModuleClassName());
345
346 m_pEventGeneratorModule->configure(
347 task.getGeneratorConfiguration().getParameters());
348
349 m_pEventGeneratorModule->prepareSubModules(
350 task.getGeneratorConfiguration().getSubModules());
351 } else {
352 throw ElemUtils::CustomException(getClassName(), __func__,
353 ElemUtils::Formatter() << "You have not provided any "
354 << EventGeneratorModule::EVENT_GENERATOR_MODULE_CLASS_NAME);
355 }
356
357 info(__func__,
358 ElemUtils::Formatter() << "Event generator module: "
359 << m_pEventGeneratorModule->toString());
360 }

406 virtual void isServiceWellConfigured() const {
407
408 if (m_pProcessModule == nullptr) {
409 throw ElemUtils::CustomException(getClassName(), __func__,
410 "Pointer to process module in null");
411 }
412
413 if (m_pKinematicModule == nullptr) {
414 throw ElemUtils::CustomException(getClassName(), __func__,
415 "Pointer to kineamtic module in null");
416 }
417
418 if (m_pRCModule == nullptr) {
419 throw ElemUtils::CustomException(getClassName(), __func__,
420 "Pointer to radiative correction module in null");
421 }
422
423 if (m_pEventGeneratorModule == nullptr) {
424 throw ElemUtils::CustomException(getClassName(), __func__,
425 "Pointer to event generator module in null");
426 }
427
428 if (m_pWriterModule == nullptr) {
429 throw ElemUtils::CustomException(getClassName(), __func__,
430 "Pointer to writer module in null");
431 }
432
433 if (m_generalConfiguration.getNEvents() == 0) {
434 throw ElemUtils::CustomException(getClassName(), __func__,
435 "Number of events to be generated is: 0");
436 }
